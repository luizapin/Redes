void bubble_sort(int list[], int n)
{
	 int c, d, t;
	 
	for (c = 0 ; c < ( n - 1 ); c++) {
		for (d = 0 ; d < n - c - 1; d++) {
			if (list[d] > list[d+1]) {
			/* Swapping */
			t = list[d];
			list[d] = list[d+1];
			list[d+1] = t;
			} // fim if
		} // fim for interno
	} // fim for externo
} // fim função bubble_sort
